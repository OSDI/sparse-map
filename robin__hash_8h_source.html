125 /**
126  * Each bucket entry has:
127  * - A value of type `ValueType`.
128  * - An integer to store how far the value of the bucket, if any, is from its ideal bucket
129  * (ex: if the current bucket 5 has the value 'foo' and `hash('foo') % nb_buckets` == 3,
130  * `dist_from_ideal_bucket()` will return 2 as the current value of the bucket is two
131  * buckets away from its ideal bucket)
132  * If there is no value in the bucket (i.e. `empty()` is true) `dist_from_ideal_bucket()` will be < 0.
133  * - A marker which tells us if the bucket is the last bucket of the bucket array (useful for the
134  * iterator of the hash table).
135  * - If `StoreHash` is true, 32 bits of the hash of the value, if any, are also stored in the bucket.
136  * If the size of the hash is more than 32 bits, it is truncated. We don't store the full hash
137  * as storing the hash is a potential opportunity to use the unused space due to the alignement
138  * of the bucket_entry structure. We can thus potentially store the hash without any extra space
139  * (which would not be possible with 64 bits of the hash).
140  */
141 template<typename ValueType, bool StoreHash>
142 class bucket_entry: public bucket_entry_hash<StoreHash> {
143  using bucket_hash = bucket_entry_hash<StoreHash>;
144

147  using distance_type = std::int_least16_t;

154  }

293 /**
294  * Internal common class used by `robin_map` and `robin_set`.
295  *
296  * ValueType is what will be stored by `robin_hash` (usually `std::pair<Key, T>` for map and `Key` for set).
297  *
298  * `KeySelect` should be a `FunctionObject` which takes a `ValueType` in parameter and returns a
299  * reference to the key.
300  *
301  * `ValueSelect` should be a `FunctionObject` which takes a `ValueType` in parameter and returns a
302  * reference to the value. `ValueSelect` should be void if there is no value (in a set for example).
303  *
304  * The strong exception guarantee only holds if the expression
305  * `std::is_nothrow_swappable<ValueType>::value && std::is_nothrow_move_constructible<ValueType>::value` is true.
306  *
307  * Behaviour is undefined if the destructor of `ValueType` throws.
308  */

343  /**
344  * Either store the hash because we are asked by the `StoreHash` template parameter
345  * or store the hash because it doesn't cost us anything in size and can be used to speed up rehash.
346  */
347  static constexpr bool STORE_HASH = StoreHash ||
348  (
349  (sizeof(tsl::detail_robin_hash::bucket_entry<value_type, true>) ==
350  sizeof(tsl::detail_robin_hash::bucket_entry<value_type, false>))
351  &&
352  (sizeof(std::size_t) == sizeof(truncated_hash_type) ||
353  is_power_of_two_policy<GrowthPolicy>::value)
354  &&
355  // Don't store the hash for primitive types with default hash.
356  (!std::is_arithmetic<key_type>::value ||
357  !std::is_same<Hash, std::hash<key_type>>::value)
358  );
359 
360  /**
361  * Only use the stored hash on lookup if we are explictly asked. We are not sure how slow
362  * the KeyEqual operation is. An extra comparison may slow things down with a fast KeyEqual.
363  */
364  static constexpr bool USE_STORED_HASH_ON_LOOKUP = StoreHash;
365 
366  /**
367  * We can only use the hash on rehash if the size of the hash type is the same as the stored one or
368  * if we use a power of two modulo. In the case of the power of two modulo, we just mask
369  * the least significant bytes, we just have to check that the truncated_hash_type didn't truncated
370  * more bytes.
371  */
372  static bool USE_STORED_HASH_ON_REHASH(size_type bucket_count) {
373  (void) bucket_count;
374  if(STORE_HASH && sizeof(std::size_t) == sizeof(truncated_hash_type)) {
375  return true;
376  }
377  else if(STORE_HASH && is_power_of_two_policy<GrowthPolicy>::value) {
378  tsl_assert(bucket_count > 0);
379  return (bucket_count - 1) <= std::numeric_limits<truncated_hash_type>::max();
380  }
381  else {
382  return false;
383  }
384  }

635  template<class InputIt>
636  void insert(InputIt first, InputIt last) {
637  if(std::is_base_of<std::forward_iterator_tag,
638  typename std::iterator_traits<InputIt>::iterator_category>::value)
639  {
640  const auto nb_elements_insert = std::distance(first, last);
641  const size_type nb_free_buckets = m_load_threshold - size();
642  tsl_assert(m_load_threshold >= size());
643 
644  if(nb_elements_insert > 0 && nb_free_buckets < size_type(nb_elements_insert)) {
645  reserve(size() + size_type(nb_elements_insert));
646  }
647  }
648 
649  for(; first != last; ++first) {
650  insert(*first);
651  }
652  }

970  const_iterator find_impl(const K& key, std::size_t hash) const {

972  distance_type dist_from_ideal_bucket = 0;
973 
974  while(dist_from_ideal_bucket <= m_buckets[ibucket].dist_from_ideal_bucket()) {

988  void erase_from_bucket(iterator pos) {
989  pos.m_iterator->clear();
990  m_nb_elements--;
991 
992  /**
993  * Backward shift, swap the empty bucket, previous_ibucket, with the values on its right, ibucket,
994  * until we cross another empty bucket or if the other bucket has a distance_from_ideal_bucket == 0.
995  *
996  * We try to move the values closer to their ideal bucket.
997  */
998  std::size_t previous_ibucket = std::size_t(std::distance(m_buckets.begin(), pos.m_iterator));
999  std::size_t ibucket = next_bucket(previous_ibucket);
1000 
1001  while(m_buckets[ibucket].dist_from_ideal_bucket() > 0) {
1002  tsl_assert(m_buckets[previous_ibucket].empty());
1003 
1004  const distance_type new_distance = distance_type(m_buckets[ibucket].dist_from_ideal_bucket() - 1);
1005  m_buckets[previous_ibucket].set_value_of_empty_bucket(new_distance, m_buckets[ibucket].truncated_hash(),
1006  std::move(m_buckets[ibucket].value()));
1007  m_buckets[ibucket].clear();
1008 
1009  previous_ibucket = ibucket;
1010  ibucket = next_bucket(ibucket);
1011  }
1012  }

1014  template<class K, class... Args>
1015  std::pair<iterator, bool> insert_impl(const K& key, Args&&... value_type_args) {
1016  const std::size_t hash = hash_key(key);
1017 
1018  std::size_t ibucket = bucket_for_hash(hash);
1019  distance_type dist_from_ideal_bucket = 0;
1020 
1021  while(dist_from_ideal_bucket <= m_buckets[ibucket].dist_from_ideal_bucket()) {
1022  if((!USE_STORED_HASH_ON_LOOKUP || m_buckets[ibucket].bucket_hash_equal(hash)) &&
1023  compare_keys(KeySelect()(m_buckets[ibucket].value()), key))
1024  {
1025  return std::make_pair(iterator(m_buckets.begin() + ibucket), false);
1026  }
1027 
1028  ibucket = next_bucket(ibucket);
1029  dist_from_ideal_bucket++;
1030  }
1031 
1032  if(grow_on_high_load()) {
1033  ibucket = bucket_for_hash(hash);
1034  dist_from_ideal_bucket = 0;
1035 
1036  while(dist_from_ideal_bucket <= m_buckets[ibucket].dist_from_ideal_bucket()) {
1037  ibucket = next_bucket(ibucket);
1038  dist_from_ideal_bucket++;
1039  }
1040  }
1041 
1042 
1043  if(m_buckets[ibucket].empty()) {
1044  m_buckets[ibucket].set_value_of_empty_bucket(dist_from_ideal_bucket, bucket_entry::truncate_hash(hash),
1045  std::forward<Args>(value_type_args)...);
1046  }
1047  else {
1048  insert_value(ibucket, dist_from_ideal_bucket, bucket_entry::truncate_hash(hash),
1049  std::forward<Args>(value_type_args)...);
1050  }
1051 
1052 
1053  m_nb_elements++;
1054  /*
1055  * The value will be inserted in ibucket in any case, either because it was
1056  * empty or by stealing the bucket (robin hood).
1057  */
1058  return std::make_pair(iterator(m_buckets.begin() + ibucket), true);

1070  truncated_hash_type hash, value_type&& value)

1092  dist_from_ideal_bucket++;

1099  void rehash_impl(size_type count) {
1100  robin_hash new_table(count, static_cast<Hash&>(*this), static_cast<KeyEqual&>(*this),
1101  get_allocator(), m_max_load_factor);
1102 
1103  const bool use_stored_hash = USE_STORED_HASH_ON_REHASH(new_table.bucket_count());
1104  for(auto& bucket: m_buckets) {
1105  if(bucket.empty()) {
1106  continue;
1107  }
1108 
1109  const std::size_t hash = use_stored_hash?bucket.truncated_hash():
1110  new_table.hash_key(KeySelect()(bucket.value()));
1111 
1112  new_table.insert_value_on_rehash(new_table.bucket_for_hash(hash), 0,
1113  bucket_entry::truncate_hash(hash), std::move(bucket.value()));
1114  }
1115 
1116  new_table.m_nb_elements = m_nb_elements;
1117  new_table.swap(*this);
1118  }
